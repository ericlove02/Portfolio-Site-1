This is a really simple PHP script used to send the Portfolio Contact form from my website to my email.

<?php
	$fname = $_POST['firstname'];
	$lname = $_POST['lastname'];
  	$email = $_POST['email'];
  	$message = $_POST['message'];

	$to = 'xxx';

	$email_subject = "Portfolio form contact";

	$email_body = "You have received a message from your portfolio site from $fname $lname.\n Here is the message:\n $message";

$headers = "From: $email \r\n";

mail($to, $email_subject, $email_body, $headers);
header('Location: http://ericlove.online/contact-recieved.html');
?>
